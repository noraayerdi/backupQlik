///$tab Calendario
// 0) Tomar min/max desde Hechos
// TempMinMaxFecha:
// LOAD
//     Floor(Min(FechaNum)) as MinFecha,
//     Floor(Max(FechaNum)) as MaxFecha
// Resident Hechos;

// LET vStart = Num(Peek('MinFecha', 0, 'TempMinMaxFecha'));
// LET vEnd   = Num(Peek('MaxFecha', 0, 'TempMinMaxFecha'));

Let vInicioCalendario = 2020;
Let vFinCalendario = Year ( Today() ) + 2;

Let vStart = Num ( Floor ( MakeDate( $(vInicioCalendario) ) ) );
 
LET vEnd = Num ( Floor ( YearEnd ( MakeDate ( $(vFinCalendario) ) ) ) );

// Salvaguardas: si vienen null => valores por defecto
// LET vStart = Alt($(vStart), Floor(YearStart(Today())));
// LET vEnd   = Alt($(vEnd),   Floor(Today()));

// Fin de año del máximo (31/12 a 00:00)
LET vEndYear = Floor(YearEnd($(vEnd)));

TRACE vStart=$(vStart);
TRACE vEnd=$(vEnd);
TRACE vEndYear=$(vEndYear);

// 1) Espina diaria continua (hasta fin de año)
NoConcatenate
CalendarSpine:
LOAD
    $(vStart) + IterNo() - 1 as FechaNum // mantener numérico
AutoGenerate 1
While $(vStart) + IterNo() - 1 <= $(vEndYear);

QuartersMap:  
MAPPING LOAD   
    RowNo() 				as Month,  
    'Q' & Ceil (RowNo()/3) 	as Quarter  
AUTOGENERATE (12); 

// 2) Tabla Calendario a partir de la espina (NO desde Hechos)
NoConcatenate
Calendario:
LOAD DISTINCT
	FechaNum											as FechaNum,
    FechaNum											as Existe_FechaNum,
    Date ( FechaNum, 'DD/MM/YYYY' )											as Fecha,
    Year(FechaNum)										as Año,
    Month(FechaNum)										as Mes,    
	Week(FechaNum) 										as Semana,
	num(Month(FechaNum)) 								as [Nº Mes],  
    Day(FechaNum) 										as Día,  
    Date(MonthStart(FechaNum), 'MMM-YYYY') 				as [Año Mes],
    Num(MonthName(FechaNum)) 							as [NumAñoMes],
    ApplyMap('QuartersMap', Month(FechaNum), Null()) 	as Trimestre, 
    Year(FechaNum) &'/'& Week(FechaNum) 				as [Año Semana],
    Day(FechaNum)&'-'& Month(FechaNum) 					as [Día Mes],
    WeekDay(FechaNum) 									as [Día Semana] 
Resident CalendarSpine;

DROP Table CalendarSpine;

	ContratoVentasID & '..'& LineaContratoVentas AS PK_ContratoLinea,
    SUM(LH_PEDIDO_CANTIDAD_KG) AS KilosServidos
Resident Hechos
WHERE ContratoVentasID>0
GROUP BY ContratoVentasID,LineaContratoVentas;

//============================================================================================================================================

// Autor:               		Mercanza

// Objetivo de la App:  		Aplicación que genera la logica de compras

Dim_Clientes:
Load
	*,
    If ( Grupo_Cliente_auxiliar <> 'No encontrado',
    	Grupo_Cliente_auxiliar,
        ClienteNombre )												as [Grupo Cliente],
        PK_Clientes													as PK_Clientes_filtro
;
LOAD
   	OrganizacionVentasID&'-'&Text(ClienteID)									as "PK_Clientes",
    OrganizacionVentasID&'-'&Text(ClienteID)									as "PK_Clientes_Propiedades",
    OrganizacionVentasID&'-'&GrupoID								as "PK_Grupo",
    GrupoCanalID													as GrupoCanalID,
	PaisID															as PaisID, 
    ClienteRegionID													as RegionID, 
    PaisID&'-'&ClienteRegionID										as PaisRegionID,
    ClienteNombre													as ClienteNombre, 
	ClienteCodigoPostal												as ClienteCodigoPostal, 
	AtencionClienteID												as AtencionClienteID, 
	AtencionCliente													as AtencionCliente, 
	ComercialID														as ComercialID,	 
	Comercial														as Comercial, 
	Inactivo														as Inactivo,
    if(Inactivo=0,'Si','No')										as VigenteCliente,
    OrganizacionVentasID											as OrganizacionVentasClientesID,
    Text(ClienteID)														as ClienteClientesID,
    GrupoID															as GrupoIDClientes,
    OrganizacionVentasID&'-'&GrupoID&'-'&Text(ClienteID)						as PK_GrupoCliente,
    GrupoCanalID&'-'&PaisID											as PK_CanalPais,
    INCO1															as Incoterm,
    ClienteNIF,
    ClienteCalle													as ClienteDireccion,
    CondicionPagoID													as CondicionPagoID,
    CondicionPago													as CondicionPago,

    NumIdentificacionFiscal											as NIF,
    ApplyMap ( 'mGrupo', GrupoID, 'No encontrado' )					as Grupo_Cliente_auxiliar,
    ClientePoblacion												as Ciudad,
    ClienteTelefono,
    LimiteCredito
    
FROM [lib://Datos:DataFiles/Dim_Clientes.qvd]
(qvd);


Concatenate ( Dim_Clientes )
Load distinct
	PK_Clientes,
    1					as ClienteSinMaestro
Resident Hechos
Where not Exists ( PK_Clientes_filtro, PK_Clientes );


Drop Field Grupo_Cliente_auxiliar;

TempUsers:
Replace LOAD
    "Name" AS [DL_DISTRIBUTION_RECIPIENT_NAME],
    "Email" AS [DL_DISTRIBUTION_EMAIL],
    SubField([Filters],',') AS [DL_DISTRIBUTION_FILTERS],
    SubField([Groups],',') AS [DL_DISTRIBUTION_GROUP_NAMES]
FROM [lib://Desarrollo:Sharepoint_COMPARTIRDATALAKE/Documentos compartidos/INFORMES/Qlik SaaS - Informes CdM_Comercial - Destinatarios.xlsx]
(ooxml, embedded labels, table is DL_DISTRIBUTION_SVC_USERS_QCS);

TempGroups:
  REPLACE LOAD
      *
    FROM [lib://Desarrollo:Sharepoint_COMPARTIRDATALAKE/Documentos compartidos/INFORMES/Qlik SaaS - Informes CdM_Comercial - Destinatarios.xlsx]
    (ooxml, embedded labels, table is DL_DISTRIBUTION_SVC_GROUPS_QCS);

let vEnabledFieldNo = FieldNumber('Enabled', 'TempGroups');

if (vEnabledFieldNo > 0) then
    DL_DISTRIBUTION_SVC_GROUPS_QCS:
    REPLACE LOAD
        "Name" AS [DL_DISTRIBUTION_GROUP_NAMES],
        "Description" AS [DL_DISTRIBUTION_GROUP_DESCRIPTION],
        "Enabled" AS [DL_DISTRIBUTION_GROUP_ENABLED]
    Resident TempGroups;
else

    DL_DISTRIBUTION_SVC_GROUPS_QCS:
    REPLACE LOAD
        "Name" AS [DL_DISTRIBUTION_GROUP_NAMES],
        "Description" AS [DL_DISTRIBUTION_GROUP_DESCRIPTION],
        1 AS [DL_DISTRIBUTION_GROUP_ENABLED]
    Resident TempGroups;
end if

Drop Table TempGroups;
