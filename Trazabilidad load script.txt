Recetas_Materiales:
LOAD
    MandanteID,
    Referencia as RecetaReferencia,
    Applymap('MapaDescripcion', Referencia) AS RecetasMateriales_RecetaDescripcion,
    PDU_Planta as RecetasMateriales_Planta,
    ListaID,
    ListaMateriales,
    NumeroLista AS RecetasMateriales_NumeroLista,
    NumeroNodo,
    TipoLista,
    ReferenciaArticuloListaMateriales AS RecetasMateriales_ComponenteReferencia,
    Applymap('MapaDescripcion', ReferenciaArticuloListaMateriales) AS RecetasMateriales_ComponenteDescripcion,
    Applymap('MapaFamilia', ReferenciaArticuloListaMateriales) AS RecetasMateriales_ComponenteFamilia,
    if(alt(num( EVALUATE(ReferenciaArticuloListaMateriales),'#0'),ReferenciaArticuloListaMateriales)='2308103371',null(), Applymap('MapaFamilia',ReferenciaArticuloListaMateriales,'') & '-' &  Applymap('MapaCodProceso', ReferenciaArticuloListaMateriales,'NA')) AS RecetasMateriales_ComponenteFamiliaProceso,
    Cantidad AS RecetasMateriales_Cantidad,
    NumeroGrupo,
    OrdenComponente,
    OrdenSustituto,
    if(OrdenSustituto='00' OR OrdenSustituto='01','','SUSTITUTO') AS RecetasMateriales_ES_SUSTITUTO,
    UnidadMedida as RecetasMateriales_Unidad,
    Validez,
    VigenteReceta,
    TipoMaterialReceta as RecetasMateriales_TipoMaterialReceta,
    TipoMaterialComponente as RecetasMateriales_TipoMaterialComponente
FROM [lib://Datos:DataFiles/Dim_MaterialesReceta.qvd]
(qvd) WHERE VigenteReceta = 1;
